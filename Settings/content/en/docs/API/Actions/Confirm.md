```kotlin
Button(text = "Confirm Example",
       onPress = listOf(
            Confirm(title = "Confirm Title",
                    message = "Confirm Message",
                    onPressOk = Alert(
                                    title = "Ok", 
                                    message = "Confirmed"),
                    onPressCancel = Alert(
                                    title = "Cancel", 
                                    message = "Cancelled"),
                    labelOk = "Accept",
                    labelCancel = "Deny"
                )
            )
        )
```
